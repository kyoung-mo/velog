<pre><code class="language-c">#define _CRT_SECURE_NO_WARNINGS
#include &lt;stdio.h&gt;

#define SIZE 50

int main() {

    printf(&quot;A=%d, Z=%d, a=%d, z=%d\n\n&quot;, 'A', 'Z', 'a', 'z');

    char arr[SIZE];
    int cnt=0;

    printf(&quot;문장 입력 : &quot;);
    fgets(arr, SIZE, stdin);

    for (int i = 0; i &lt; SIZE; i++) {
        if (arr[i] &gt;= 65 &amp;&amp; arr[i] &lt;= 90) {
            arr[i] += 32;
            cnt++;
        }
        printf(&quot;%c&quot;, arr[i]);
        if (arr[i] == '\0') break;
    }
    printf(&quot;\n바뀐 문자 수 : %d&quot;, cnt);

    return 0;
}</code></pre>

<pre><code class="language-c">#define _CRT_SECURE_NO_WARNINGS
#include &lt;stdio.h&gt;

void swap(double* x, double* y);

int main() {

    double a, b, c;
    double* pa, * pb, * pc;

    pa = &amp;a;
    pb = &amp;b;
    pc = &amp;c;

    printf(&quot;실수값 3개 입력 : &quot;);
    scanf(&quot;%lf %lf %lf&quot;, &amp;a, &amp;b, &amp;c);

    swap(pa, pb);
    swap(pa, pc);
    swap(pb, pc);

    printf(&quot;정렬된 값 출력 : %.1lf %.1lf %.1lf&quot;, a, b, c);

    return 0;
}

void swap(double* x, double* y) {
    double temp;

    if (*x &lt; *y) {
        temp = *x;
        *x = *y;
        *y = temp;
    }
}</code></pre>
